    <h2>Asymmetric Label Dashboard</h2>

    <p style="font-size: 13px; color: #666; margin-bottom: 20px;">Splits label: Left 25% for QR | Right 75% for Content Text.</p>

    

    <button class="btn-blue" id="actionBtn" onclick="pairTscPrinter()">Connect Alpha-30RB (Bluetooth)</button>

    <input type="text" id="upcBox" placeholder="Scan UPC Code Here..." onkeypress="handleUpcScan(event)" disabled>

    

    <div id="status">Waiting for Bluetooth Printer Link...</div>

<script>

    // ⚠️ PASTE YOUR GOOGLE WEB APP MACRO URL LINK INSIDE THE QUOTES BELOW:

    const GOOGLE_API_LINK = "https://script.google.com/a/macros/zomato.com/s/AKfycbyG7d8zZvM3pIyXMAS1eH8t5JC8k2G3uMH3PxnIIbbCtdxByXxCAas6aFvEzYbabCEc/exec"; 

    

    let tscBleCharacteristic = null;


    // 1. Establish direct pairing connection using Google Chrome Web Bluetooth features

    async function pairTscPrinter() {

        const log = document.getElementById('status');

        try {

            log.innerText = "Searching for your TSC Printer...";

            const selectedDevice = await navigator.bluetooth.requestDevice({

                filters: [{ namePrefix: 'Alpha-30' }, { namePrefix: 'TSC' }],

                optionalServices: ['00001101-0000-1000-8000-00805f9b34fb'] // SPP Profile

            });

            

            log.innerText = "Connecting to device server cards...";

            const openGatt = await selectedDevice.gatt.connect();

            const targetService = await openGatt.getPrimaryService('00001101-0000-1000-8000-00805f9b34fb');

            tscBleCharacteristic = await targetService.getCharacteristic('00001101-0000-1000-8000-00805f9b34fb');

            

            log.innerText = "Printer Paired & Ready!";

            log.className = "ready";

            

            document.getElementById('upcBox').disabled = false;

            document.getElementById('upcBox').focus();

            document.getElementById('actionBtn').style.display = 'none';

        } catch (failReason) {

            log.innerText = "Bluetooth Setup Error: " + failReason;

            console.error(failReason);

        }

    }


    // 2. Catch Scanner Input Event Actions

    async function handleUpcScan(event) {

        if (event.key === 'Enter') {

            const upcQuery = document.getElementById('upcBox').value.trim();

            const log = document.getElementById('status');

            document.getElementById('upcBox').value = ''; // Flush input field immediately

            

            if (!upcQuery) return;

            

            log.innerText = `Searching records for UPC: ${upcQuery}...`;

            log.className = "";

            

            try {

                const networkResponse = await fetch(GOOGLE_API_LINK);

                const databaseRows = await networkResponse.json();

                

                // Match the scanned UPC value against Column A data

                const matchedRecord = databaseRows.find(entry => entry.upc === upcQuery);

                

                if (matchedRecord) {

                    log.innerText = "Match located! Dispatched to printer module...";

                    await executeAsymmetricPrint(matchedRecord.qrData, matchedRecord.label1, matchedRecord.label2);

                } else {

                    log.innerText = `UPC "${upcQuery}" could not be found in your Google Sheet.`;

                }

            } catch (networkError) {

                log.innerText = "Data Retrieval Error: " + networkError;

            }

        }

    }


    // 3. Compile layout geometry variables and write safely across Bluetooth communication card ports

    async function executeAsymmetricPrint(qrString, label1String, label2String) {

        // Precise structural ZPL template coordinates requested

        const structuredZpl = 

            "^XA\n" +

            "^PW576\n" + 

            "^LL406\n" + 

            "^FO15,15^GB546,376,4^FS\n" +                // Border Container

            "^FO160,15^GB0,376,3^FS\n" +                 // Vertical 25% Grid Separator Line

            "^FO160,203^GB401,0,3^FS\n" +                // Horizontal 50/50 Right Splitter Line

            "^FO30,80^BQN,2,5^FDQA," + qrString + "^FS\n" + // QR Location (Left)

            "^FO180,35^A0N,20,20^FDLABEL 1:^FS\n" +

            "^FO180,65^A0N,26,24^FB360,4,0,L^FD" + label1String + "^FS\n" + // Label 1 Top Text Block

            "^FO180,223^A0N,20,20^FDLABEL 2:^FS\n" +

            "^FO180,253^A0N,26,24^FB360,4,0,L^FD" + label2String + "^FS\n" + // Label 2 Bottom Text Block

            "^XZ\n";


        // Transmit string mapped to raw data bytes

        const encoderObj = new TextEncoder();

        const binaryPayload = encoderObj.encode(structuredZpl);

        

        // Safety protocol: Stream data chunks over 128 byte bounds to ensure mobile devices don't drop text

        const packetBound = 128;

        try {

            for (let position = 0; position < binaryPayload.byteLength; position += packetBound) {

                const chunkSlice = binaryPayload.slice(position, position + packetBound);

                await tscBleCharacteristic.writeValue(chunkSlice);

                await new Promise(done => setTimeout(done, 40)); // Short 40ms channel buffer rest

            }

            document.getElementById('status').innerText = "Label Output Success!";

            document.getElementById('status').className = "ready";

        } catch (printFailure) {

            document.getElementById('status').innerText = "Hardware Writing Interrupted: " + printFailure;

        }

    }

</script>
